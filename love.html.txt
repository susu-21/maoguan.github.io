        // 随机生成彩灯装饰
        const tree = document.querySelector('.tree');
        const colors = ['#ff0000', '#00ff00', '#0000ff', '#ffff00', '#ff00ff', '#00ffff'];
        const ornamentCount = 15;

        for (let i = 0; i < ornamentCount; i++) {
            const ornament = document.createElement('div');
            ornament.classList.add('ornament');
            ornament.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
            ornament.style.left = `${Math.random() * 100}%`;
            ornament.style.top = `${Math.random() * 80}%`;
            tree.appendChild(ornament);
        }
